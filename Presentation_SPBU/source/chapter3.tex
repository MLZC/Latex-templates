\subsection{Multiline Equation}
\begin{frame}
    \frametitle{Multiline Equation}
    Aligning several equations with no numbers  \cite{oetiker1995not}
    .
    \begin{align*}
        x&=y           &  w &=z              &  a&=b+c\\
        2x&=-y         &  3w&=\frac{1}{2}z   &  a&=b\\
        -4 + 5x&=2+y   &  w+2&=-1+w          &  ab&=cb
    \end{align*}
    Other way.
    \begin{equation}
    \begin{aligned}
        x&=y           &  w &=z              &  a&=b+c\\
        2x&=-y         &  3w&=\frac{1}{2}z   &  a&=b\\
        -4 + 5x&=2+y   &  w+2&=-1+w          &  ab&=cb
    \end{aligned}
\end{equation}
\end{frame}
